    @export annotation

    When using the @export annotation, be certain that you use the --generate_exports compiler flag.

    goog.provide('app.start');

/**
 * Make {@code app.start} accessible after {@code ADVANCED_OPTIMIZATIONS}.
 * @export
 */
app.start = function() {
    // ...
};
